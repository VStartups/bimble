fn check_gcc() {
    let out = Command::new("gcc")
        .arg("--version")
        .output()
        .unwrap_or_else(|_| {
            eprintln!("gcc is not installed or not found in the system's PATH.");
            exit(1);
        });

    if !out.status.success() {
        eprintln!("gcc is installed, but there was an issue running the command.");
        exit(1);
    }
}
